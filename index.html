  url: "<?php echo base_url() . $this->modul_name . '/getDetAngsuran'; ?>",
  type: 'POST', //method: 'POST',

  angsuran_id: angsuran_id
  },

Ajax call codeigniter function json.

            
$.ajax({
  url: "<?php echo base_url() . $this->modul_name . '/getDetAngsuran'; ?>",
  type: 'POST',
  dataType: "json",
  data: {
    angsuran_id: angsuran_id
  },
  success: function(res) {
    alert(res.angsuran_id);
    $('#id_angsuran').html(res);
  }
});

Setting php.ini

Cari file php.ini

Cari:
post_max_size = 8M
upload_max_filesize = 2M
max_execution_time = 30
max_input_time = 60
memory_limit = 8M


Ubah nilai menjadi:
post_max_size = 750M
upload_max_filesize = 750M
max_execution_time = 5000
max_input_time = 5000
memory_limit = 1000M

Cari:
short_open_tag=Off

Ubah nilai menjadi:
short_open_tag=On

kemudian restart apache
