                  url: "<?php echo base_url() . $this->modul_name . '/getDetAngsuran'; ?>",
                  type: 'POST', //method: 'POST',

Ajax call codeigniter function json.

            
                $.ajax({
                  url: "<?php echo base_url() . $this->modul_name . '/getDetAngsuran'; ?>",
                  type: 'POST',
                  dataType: "json",
                  data: {
                    angsuran_id: angsuran_id
                  },
                  success: function(res) {
                    alert(res.angsuran_id);
                    $('#id_angsuran').html(res);
                  }
                });
