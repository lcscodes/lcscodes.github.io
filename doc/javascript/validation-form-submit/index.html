            //script
            function formValidation() {
				var table = document.getElementById("tableId");
				var tbodyRowCount = table.tBodies[0].rows.length; // 3
				if (tbodyRowCount == 0) {
					alert('Rincian Rekening Tidak Ada');
					return false;
				} else {
					return true;
				}
			}

            //form attribute
            <form class="form-horizontal" enctype="multipart/form-data" action="" method="post" name="myform" id="myform" onSubmit="return formValidation();"></form>


            //untuk validasi hapus
            <script>
                function formValidation(idkey) {
                    $.ajax({
                        url: "<?php echo base_url() . $this->modul_name . '/cekBeforeDelete'; ?>",
                        type: 'POST', //method: 'POST',
                        async: false,
                        data: {
                            idkey: idkey
                        },
                        success: function(res) {
                            resx = res;
                        }
                    });
                    if (resx > 0) {
                        alert('barang tidak bisa dihapus karena sudah dipakai untuk transaksi..');
                        return false;
                    } else {
                        var result = confirm("Yakin akan dihapus?");
                        if (result) {
                            alert('hapus berhasil');
                            return true;
                        } else {
                            return false;
                        }
                    }
                }
                </script>

                //form attribute
                <a href=" <?php echo base_url() . $this->modul_name . '/delete/' . $row->$key; ?>" onclick="return formValidation( <?php echo $row->$key; ?>)"> <i class="fa fa-trash"> </i>
                </a>
